-- Criação da ScreenGui
local screenGui = Instance.new("ScreenGui")
screenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

-- Frame principal
local mainFrame = Instance.new("Frame")
mainFrame.Size = UDim2.new(0, 300, 0, 200)
mainFrame.Position = UDim2.new(0.5, -150, 0.5, -100)
mainFrame.BackgroundColor3 = Color3.new(0.2, 0.2, 0.2)
mainFrame.Parent = screenGui

-- Título da GUI
local titleLabel = Instance.new("TextLabel")
titleLabel.Text = "Configurações da Killaura"
titleLabel.Size = UDim2.new(1, 0, 0, 30)
titleLabel.TextColor3 = Color3.new(1, 1, 1)
titleLabel.BackgroundColor3 = Color3.new(0.3, 0.3, 0.3)
titleLabel.Parent = mainFrame

-- Botão On/Off
local toggleButton = Instance.new("TextButton")
toggleButton.Text = "Killaura: OFF"
toggleButton.Size = UDim2.new(1, 0, 0, 30)
toggleButton.Position = UDim2.new(0, 0, 0, 40)
toggleButton.BackgroundColor3 = Color3.new(0.5, 0, 0)
toggleButton.TextColor3 = Color3.new(1, 1, 1)
toggleButton.Parent = mainFrame

-- Caixa de texto para o dano
local damageLabel = Instance.new("TextLabel")
damageLabel.Text = "Dano:"
damageLabel.Size = UDim2.new(0.4, 0, 0, 30)
damageLabel.Position = UDim2.new(0, 0, 0, 80)
damageLabel.TextColor3 = Color3.new(1, 1, 1)
damageLabel.BackgroundTransparency = 1
damageLabel.Parent = mainFrame

local damageInput = Instance.new("TextBox")
damageInput.Text = "25"
damageInput.Size = UDim2.new(0.6, 0, 0, 30)
damageInput.Position = UDim2.new(0.4, 0, 0, 80)
damageInput.TextColor3 = Color3.new(1, 1, 1)
damageInput.BackgroundColor3 = Color3.new(0.3, 0.3, 0.3)
damageInput.Parent = mainFrame

-- Caixa de texto para a distância
local distanceLabel = Instance.new("TextLabel")
distanceLabel.Text = "Distância Máx.:"
distanceLabel.Size = UDim2.new(0.4, 0, 0, 30)
distanceLabel.Position = UDim2.new(0, 0, 0, 120)
distanceLabel.TextColor3 = Color3.new(1, 1, 1)
distanceLabel.BackgroundTransparency = 1
distanceLabel.Parent = mainFrame

local distanceInput = Instance.new("TextBox")
distanceInput.Text = "10"
distanceInput.Size = UDim2.new(0.6, 0, 0, 30)
distanceInput.Position = UDim2.new(0.4, 0, 0, 120)
distanceInput.TextColor3 = Color3.new(1, 1, 1)
distanceInput.BackgroundColor3 = Color3.new(0.3, 0.3, 0.3)
distanceInput.Parent = mainFrame

local killauraEnabled = false
local damageAmount = tonumber(damageInput.Text)
local maxDistance = tonumber(distanceInput.Text)

toggleButton.MouseButton1Click:Connect(function()
    killauraEnabled = not killauraEnabled
    toggleButton.Text = killauraEnabled and "Killaura: ON" or "Killaura: OFF"
    toggleButton.BackgroundColor3 = killauraEnabled and Color3.new(0, 0.5, 0) or Color3.new(0.5, 0, 0)
end)

damageInput.FocusLost:Connect(function()
    damageAmount = tonumber(damageInput.Text) or damageAmount
end)

distanceInput.FocusLost:Connect(function()
    maxDistance = tonumber(distanceInput.Text) or maxDistance
end)

-- Script de ataque
while true do
    if killauraEnabled then
        for _, npc in pairs(game.Workspace:GetChildren()) do
            if npc:IsA("Model") and npc:FindFirstChild("Humanoid") and npc.Name == "dummy" then
                local character = game.Players.LocalPlayer.Character
                local distance = (character.HumanoidRootPart.Position - npc.HumanoidRootPart.Position).magnitude
                if distance <= maxDistance then
                    npc.Humanoid:TakeDamage(damageAmount)
                    print("Atacando Boss com killaura")
                end
            end
        end
    end
    wait(1) -- Intervalo entre ataques
end
